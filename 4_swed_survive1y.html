Run categorical model on cluster

bsub -a openmpi -q mpi-long -W 120:00 -n 16,20 -R span[hosts=1] mpirun.lsf R --slave -f "/usr/users/rarslan/updated_data/4_swed_survive1y_subset.R" drew subset like swed.2 = sample_level2(na.omit(swed.2[, list(survive1y, idParents, dependent_sibs_f5y, maternalage.factor,paternalloss, maternalloss, nr.siblings, birth.cohort, male, paternalage.mean, paternalage.factor, paternalage.diff)]),"idParents",100000)

setwd("/usr/users/rarslan/updated_data/")
library(doMPI)
cl <- startMPIcluster(verbose=T,workdir="/usr/users/rarslan/updated_data/")
registerDoMPI(cl)
Model = foreach(i=1:6,.options.mpi = list(seed=1337) ) %dopar% {
    library(MCMCglmm);library(data.table)
    setwd("/usr/users/rarslan/updated_data/")
        source("0__helpers.R")
    load("swed2_subset.rdata")
    
    samples = 500
    thin = 1000; burnin = 1500000
    nitt = samples * thin + burnin 

    prior <- list(
        R=list(V=diag(1), nu=1.002), 
        G=list(G1=list(V=diag(1), nu=1, alpha.mu=c(0), alpha.V=diag(1)*1000))
    )
    
    start <- list(
        liab=c(rnorm( nrow(swed.2) )), 
        R = list(R1 = rIW(diag(1), 10)),
        G = list(G1 = rIW(diag(1), 10 ))
    )

    ( m1 = MCMCglmm( survive1y ~ birth.cohort + male + paternalage.mean + paternalage.factor,
                        random=~idParents,
                        family="categorical",
                        start = start,
                        prior = prior,
                        data=swed.2, 
                        pr = F, saveX = T, saveZ = T,
                        nitt=nitt,thin=thin,burnin=burnin)
    )
}

saveRDS(Model,file = "coefs/swed_survive1y_subset.rds")

closeCluster(cl)
mpi.quit()
