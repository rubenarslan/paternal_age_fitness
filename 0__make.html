the rendering function for Rmds and md

render_web = function(file, auto_tab = T, include_top = "library/general_header.html") { 
    library(stringr); library(rmarkdown); library(dplyr); library(stringi)
    file %>% 
        human_readable_filename() -> 
        title
    pops = c("krmh","ddb","rpqa","swed")
    pops = pops[file %contains% pops]

    if(length(pops) == 1) include_top = paste0("library/",pops,"_header.html")
    pandoc_args = list(paste0("--metadata=title:\"",title,"\""))
    if(auto_tab) pandoc_args[[2]] = "--metadata=autotab:1"
    output_format = html_document(
        self_contained = F,
        template = "../paternal_age_genealogies_pages/library/website_theme.html",
        lib_dir="../paternal_age_genealogies_pages/library/",
        mathjax = NULL,
        keep_md = T,
        includes = includes(before_body = include_top),
        pandoc_args = pandoc_args
    )
    output_format$knitr$opts_chunk$error = TRUE
    
    render(input = file, 
        output_format = output_format, 
        envir = new.env(), 
        output_dir = "../paternal_age_genealogies_pages/"
    )
}
