Run zero-altered Poisson model on cluster

this script runs a zero-altered Poisson model on our scientific computing cluster using the following command bsub -a openmpi -q mpi -W 48:00 -n 16 -R span[hosts=1] mpirun.lsf R --slave -f "/usr/users/rarslan/updated_data/4_rpqa_all_children_surviving5y_linear_grandparents.R"

setwd("/usr/users/rarslan/updated_data/")
library(doMPI)
cl <- startMPIcluster(verbose=T,workdir="/usr/users/rarslan/updated_data/")
registerDoMPI(cl)
Children = foreach(i=1:clusterSize(cl),.options.mpi = list(seed=1337) ) %dopar% {
    library(MCMCglmm);library(data.table)
    setwd("/usr/users/rarslan/updated_data/")
        source("0__helpers.R")
    load("rpqa_grandparents.rdata")
    rpqa.1 = na.omit(rpqa.1[byear <= 1740, list(children.surviving5y, dependent_sibs_f5y, maternalage.factor, paternalloss, maternalloss, nr.siblings, birth.cohort, male, urban, paternalage.mean, paternalage.diff, idParents, maternal.grandpaternalage.diff, maternal.grandpaternalage.mean, paternal.grandpaternalage.diff, paternal.grandpaternalage.mean, idMaternalGrandparents, idPaternalGrandparents)])
    
    samples = 1000
    thin = 100; burnin = 10000
    nitt = samples * thin + burnin

    prior <- list(
        R=list(V=diag(2), nu=1.002, fix=2), 
        G=list(G1=list(V=diag(2), nu=1, alpha.mu=c(0,0), alpha.V=diag(2)*1000),
                     G2=list(V=diag(2), nu=1, alpha.mu=c(0,0), alpha.V=diag(2)*1000),
                     G3=list(V=diag(2), nu=1, alpha.mu=c(0,0), alpha.V=diag(2)*1000))
    )
    
    start <- list(
        liab=c(rnorm( nrow(rpqa.1)*2 )), 
        R = list(R1 = rIW(diag(2), 10 , fix = 2)),
        G = list(G1 = rIW(diag(2), 10 ),
                         G2 = rIW(diag(2), 10 ),
                         G3 = rIW(diag(2), 10 ))
    )

    ( m1 = MCMCglmm( children.surviving5y ~ trait * (birth.cohort + male + urban + paternalage.mean + paternalage.diff + maternal.grandpaternalage.diff + maternal.grandpaternalage.mean + paternal.grandpaternalage.diff + paternal.grandpaternalage.mean) - 1,
                        rcov = ~idh(trait):units,
                        random= ~ idh(trait):idParents + idh(trait):idMaternalGrandparents + idh(trait):idPaternalGrandparents,
                        family = "zapoisson",
                        start = start,
                        prior = prior,
                        data = rpqa.1, 
                        pr = F, saveX = T, saveZ = T,
                        nitt = nitt, thin = thin, burnin = burnin)
    )
}

saveRDS(Children,file = "coefs/rpqa_Children_surviving5y_all_linear_grandparents.rds")

closeCluster(cl)
mpi.quit()
